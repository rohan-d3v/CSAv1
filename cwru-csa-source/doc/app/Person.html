def can_change_name?
  return true if is_admin

  current_game = Game.current
  current_registration = registrations.where(:game_id => current_game.id).first

  return current_registration.blank? || !current_game.has_begun?
end

def legal_to_sign_waiver?
  return true unless date_of_birth.present?  # assume > 18 until known otherwise
  ((Date.today - date_of_birth).days / 1.year) >= 18
end
