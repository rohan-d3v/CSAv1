def profile_photo_url(person, size=32, default=image_path('http://casehvz.com/assets/gravatar_default.png'))
  ["http://www.gravatar.com/avatar/",
    Digest::MD5.hexdigest(person.caseid + '@case.edu'),
    "?s=#{size}&d=#{u default}"
  ].join
end
